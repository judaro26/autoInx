function renderOrders(orders) { console.log("=== RENDER ORDERS DEBUG ==="); console.log("Number of orders:", orders.length); console.log("Full orders data:", orders); const container = document.getElementById('ordersList'); const ordersContainerEl = document.getElementById('ordersContainer'); console.log("Container element:", container); console.log("Orders container element:", ordersContainerEl); if (!container) { console.error("ERROR: ordersList container not found!"); return; } if (!ordersContainerEl) { console.error("ERROR: ordersContainer element not found!"); return; } // --- 1. Render Logic (Empty State) --- if (orders.length === 0) { console.log("Rendering empty state"); container.innerHTML = `

Aún no tienes pedidos

Cuando hagas tu primera compra, aparecerá aquí

Ir a la Tienda
`; } // --- 2. Render Logic (Orders List) --- else { console.log("Rendering", orders.length, "orders"); // Check if order data has required fields orders.forEach((order, index) => { console.log(`Order ${index}:`, { id: order.id, totalCents: order.totalCents, items: order.items, itemsCount: order.items ? order.items.length : 0, createdAt: order.createdAt, isPaid: order.isPaid, status: order.status }); }); const ordersHTML = orders.map((order, idx) => { const statusColors = { 'Delivered': 'bg-emerald-100 text-emerald-800 border-emerald-200', 'Shipped': 'bg-blue-100 text-blue-800 border-blue-200', 'Processing': 'bg-indigo-100 text-indigo-800 border-indigo-200', 'Cancelled': 'bg-red-100 text-red-800 border-red-200', 'Pending': 'bg-yellow-100 text-yellow-800 border-yellow-200', 'Manually Created': 'bg-purple-100 text-purple-800 border-purple-200', }; const badge = statusColors[order.status] || 'bg-gray-100 text-gray-800 border-gray-200'; // Ensure items array exists const items = order.items || []; const isPaid = order.isPaid || false; const totalCents = order.totalCents || 0; const deliveryAddress = order.deliveryAddress || 'No especificada'; const createdAt = order.createdAt || order.timestamp; return `

Pedido #${order.id ? order.id.slice(0, 10) : 'N/A'}

${formatTime(createdAt)}

${isPaid ? 'PAGADO' : 'PENDIENTE'} ${order.status || 'PENDING'}

Total Pagado

${formatPrice(totalCents)}

Dirección de Envío

${deliveryAddress}

Artículos

${items.length}

Productos comprados:

    ${items.map(item => `
  • ${item.quantity || 1}x ${item.name || 'Producto'} ${formatPrice((item.price || 0) * (item.quantity || 1))}
    • `).join('')}
`; }).join(''); console.log("Generated HTML length:", ordersHTML.length); console.log("First 500 chars of HTML:", ordersHTML.substring(0, 500)); container.innerHTML = ordersHTML; } // --- 3. Make the container visible --- console.log("Making container visible..."); ordersContainerEl.classList.remove('hidden'); ordersContainerEl.classList.remove('opacity-0'); // Force a reflow to trigger CSS animation void ordersContainerEl.offsetWidth; console.log("Render complete!"); }
